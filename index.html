// Saved by Goalkeeper! return; } } // Check Goal Line Boundary crossing if it bypassed goalkeeper save hitbox if (ball.y <= goal.y + goal.h && ball.y >= goal.y - 10) { if (ball.x >= goal.x && ball.x <= goal.x + goal.w) { processResult(true);
